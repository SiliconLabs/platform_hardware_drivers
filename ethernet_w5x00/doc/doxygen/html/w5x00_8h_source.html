460#define __GP_REGISTER8(name, address) \
461 static inline uint16_t w5x00_write##name(uint8_t data) { \
462 uint8_t bw = data; \
463 return w5x00_write(address, &bw, 1); \
464 } \
465 static inline uint8_t w5x00_read##name() { \
466 uint8_t br; \
467 w5x00_read(address, &br, 1); \
468 return br; \
469 }
470#define __GP_REGISTER16(name, address) \
471 static inline void w5x00_write##name(uint16_t _data) { \
472 uint8_t buf[2]; \
473 buf[0] = _data >> 8; \
474 buf[1] = _data & 0xFF; \
475 w5x00_write(address, buf, 2); \
476 } \
477 static inline uint16_t w5x00_read##name() { \
478 uint8_t buf[2]; \
479 w5x00_read(address, buf, 2); \
480 return (buf[0] << 8) | buf[1]; \
481 }
482#define __GP_REGISTER_N(name, address, size) \
483 static inline uint16_t w5x00_write##name(const uint8_t *_buff) { \
484 return w5x00_write(address, _buff, size); \
485 } \
486 static inline uint16_t w5x00_read##name(uint8_t *_buff) { \
487 return w5x00_read(address, _buff, size); \
488 }
