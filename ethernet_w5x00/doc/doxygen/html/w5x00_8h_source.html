473#define __GP_REGISTER8(name, address) \
474 static inline uint16_t w5x00_write##name(uint8_t data) { \
475 uint8_t bw = data; \
476 return w5x00_write(address, &bw, 1); \
477 } \
478 static inline uint8_t w5x00_read##name() { \
479 uint8_t br; \
480 w5x00_read(address, &br, 1); \
481 return br; \
482 }
483#define __GP_REGISTER16(name, address) \
484 static inline void w5x00_write##name(uint16_t _data) { \
485 uint8_t buf[2]; \
486 buf[0] = _data >> 8; \
487 buf[1] = _data & 0xFF; \
488 w5x00_write(address, buf, 2); \
489 } \
490 static inline uint16_t w5x00_read##name() { \
491 uint8_t buf[2]; \
492 w5x00_read(address, buf, 2); \
493 return (buf[0] << 8) | buf[1]; \
494 }
495#define __GP_REGISTER_N(name, address, size) \
496 static inline uint16_t w5x00_write##name(const uint8_t *_buff) { \
497 return w5x00_write(address, _buff, size); \
498 } \
499 static inline uint16_t w5x00_read##name(uint8_t *_buff) { \
500 return w5x00_read(address, _buff, size); \
501 }
